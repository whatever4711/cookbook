\begin{recipe}
[% 
    preparationtime = {\unit[15]{min}},
    portion = {\portion{4}},
    source = {Lieblingsrezept von Udo (original von chefkoch.de)}
]
{Puten-Gemüse-Curry}
  
	\ingredients{%
        500 g	& 	Putenbrust \\
		2	& 	Paprikaschoten, rot \\
		6 &	Karotten \\
		2	&	Zwiebeln\\
		6	&	große Kartoffeln \\
		6 Tassen	&	Gemüsebrühe \\
		2 Tassen & Milch oder Kokosmilch \\
		4 EL & Currypulver\\
		 ~ & Öl\\
		 ~ & Salz und Pfeffer\\
		 ~ & Zucker\\
		~ & Ingwerpulver\\
    }
    
    \preparation{%
    \newline
       \step Putenbrust, Paprika, Karotten und Kartoffeln klein schneiden. Die Zwiebel in kleine Würfel schneiden und in Öl anbraten. Gemüse und Putenbrust dazugeben und kurz mitrösten. 
       \step Mit Suppe und Milch aufgießen. Nach Geschmack würzen, alles schmoren bis alle Zutaten durchgegart sind.
       \vspace{3cm}
    }
    
    \suggestion[Servierbeispiel]
    {%
       Mit Reis servieren.
    }
\end{recipe}
